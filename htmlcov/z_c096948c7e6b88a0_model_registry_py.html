198 @public 

199 @usage.method_logger 

200 def hf_download( 

201 self, 

202 hugging_face_model_id: str, 

203 hf_token: str | None = None, 

204 selected_formats: list[str] | None = None, 

205 selected_variants: list[str] | None = None, 

206 selected_filenames: list[str] | None = None, 

207 timeout: int = 36000, 

208 poll_interval: int = 5, 

209 ) -> model.Model: 

210 """Import a model from HuggingFace into this Model Registry. 

211 

212 Kicks off the asynchronous server-side download, polls until the import 

213 reaches a terminal status, and returns the registered model on success. 

214 

215 ```python 

216 import hopsworks 

217 

218 project = hopsworks.login() 

219 mr = project.get_model_registry() 

220 

221 # Public model — minimal call. 

222 m = mr.hf_download("Qwen/Qwen2.5-0.5B") 

223 

224 # Gated / private — supply an HF access token. 

225 m = mr.hf_download("meta-llama/Llama-3.2-1B", hf_token="hf_xxx") 

226 

227 # Multi-quant repo — pick exactly one variant. 

228 m = mr.hf_download( 

229 "unsloth/Kimi-K2.6-GGUF", selected_variants=["UD-Q4_K_XL"] 

230 ) 

231 ``` 

232 

233 Parameters: 

234 hugging_face_model_id: HF repo id (``owner/repo``) or full HuggingFace URL. 

235 hf_token: Optional HuggingFace access token; required for gated/private repos. 

236 selected_formats: Weight-format keys to download (``safetensors``, ``pytorch``, 

237 ``tensorflow``, ``flax``, ``gguf``, ``onnx``, ``openvino``, ``coreml``, 

238 ``tflite``). When omitted, the backend picks one with the precedence 

239 safetensors > pytorch > tensorflow > flax > … so multi-format repos don't 

240 pull every copy of the same weights. 

241 selected_variants: GGUF / unsloth quantisation tags to keep 

242 (``Q4_K_M``, ``UD-Q4_K_XL``, ``BF16``, …). Acts as a sub-filter — files 

243 with a quant tag are only downloaded if their tag is in the list. Files 

244 without a tag (root README, config, …) are unaffected. 

245 selected_filenames: Explicit per-file allowlist. When non-empty this 

246 overrides ``selected_formats`` / ``selected_variants`` and the backend 

247 downloads exactly these paths. 

248 timeout: Maximum seconds to wait for the import to reach a terminal state. 

249 poll_interval: Seconds between status polls. 

250 

251 Returns: 

252 The newly registered model entity. 

253 

254 Raises: 

255 hopsworks.client.exceptions.HuggingFaceImportException: If the backend 

256 fails the import with a known terminal error. The exception's 

257 ``error_code`` field carries the stable code (``auth_required``, 

258 ``model_not_found``, ``not_found_or_auth_required``, ``no_disk_space``, 

259 ``fetch_failed``, etc.). 

260 hopsworks.client.exceptions.RestAPIError: For unexpected server errors 

261 that are not part of the HF import error set. 

262 """ 

263 # Map known synchronous backend errors (returned by POST /huggingface and 

264 # /inspect on the resource) to the typed exception so the caller doesn't 

265 # need to introspect a raw RestAPIError. 

266 try: 

267 start = self._huggingface_api.start_import( 

268 self.model_registry_id, 

269 hugging_face_model_id, 

270 hf_token=hf_token, 

271 selected_formats=selected_formats, 

272 selected_variants=selected_variants, 

273 selected_filenames=selected_filenames, 

274 ) 

275 except RestAPIError as e: 

276 self._raise_hf_exception_from_rest_error(e) 

277 raise # pragma: no cover 

278 

279 job_id = (start or {}).get("jobId") if isinstance(start, dict) else None 

280 if not job_id: 

281 raise HuggingFaceImportException( 

282 HuggingFaceImportException.FETCH_FAILED, 

283 "Backend did not return a jobId for the HuggingFace import.", 

284 ) 

285 

286 print( 

287 f"Importing '{hugging_face_model_id}' (job {job_id}). " 

288 f"Polling every {poll_interval}s for up to {timeout}s." 

289 ) 

290 

291 deadline = time.time() + timeout 

292 last_status: dict = {} 

293 pbar: tqdm | None = None 

294 last_completed = 0 

295 try: 

296 while time.time() < deadline: 

297 last_status = self._huggingface_api.get_status( 

298 self.model_registry_id, job_id 

299 ) 

300 status = (last_status or {}).get("status") 

301 total = int(last_status.get("totalFiles") or 0) 

302 done = int(last_status.get("completedFiles") or 0) 

303 current_file = last_status.get("currentFile") 

304 

305 # Lazy-init the progress bar once the backend reports a file count. 

306 if pbar is None and total > 0: 

307 pbar = tqdm(total=total, unit="file", desc="HF import") 

308 if pbar is not None: 

309 if done > last_completed: 

310 pbar.update(done - last_completed) 

311 last_completed = done 

312 if current_file: 

313 pbar.set_postfix_str(current_file, refresh=False) 

314 

315 if status == "COMPLETED": 

316 if pbar is not None: 

317 if total > last_completed: 

318 pbar.update(total - last_completed) 

319 pbar.close() 

320 print( 

321 f"✔ HuggingFace import succeeded: " 

322 f"'{hugging_face_model_id}' downloaded " 

323 f"({done}/{total} files)." 

324 ) 

325 return self._resolve_imported_model( 

326 hugging_face_model_id, last_status 

327 ) 

328 if status in ("FAILED", "CANCELLED"): 

329 if pbar is not None: 

330 pbar.close() 

331 error = last_status.get("error") or "unknown_error" 

332 raise self._build_hf_exception(error) 

333 time.sleep(poll_interval) 

334 finally: 

335 if pbar is not None: 

336 pbar.close() 

337 

338 # Timed out — best-effort cancel so the partial dir doesn't linger. 

339 with contextlib.suppress(Exception): 

340 self._huggingface_api.cancel(self.model_registry_id, job_id, cleanup=True) 

341 raise HuggingFaceImportException( 

342 HuggingFaceImportException.FETCH_FAILED, 

343 f"HuggingFace import {hugging_face_model_id} did not finish within " 

344 f"{timeout}s. Last status: {last_status}", 

345 )

347 @staticmethod 

348 def _raise_hf_exception_from_rest_error(err: RestAPIError) -> None: 

349 """If a RestAPIError carries one of the known HF backend codes, re-raise. 

350 

351 Promotes the error to :class:`HuggingFaceImportException`. Returns 

352 without raising for unknown codes so the caller can re-raise the 

353 original RestAPIError untouched. 

354 """ 

355 body = err.response.json() if hasattr(err, "response") else {} 

356 if not isinstance(body, dict): 

357 body = {} 

358 code = body.get("error") or body.get("errorCode") 

359 message = body.get("message") or body.get("usrMsg") 

360 known = { 

361 HuggingFaceImportException.AUTH_REQUIRED, 

362 HuggingFaceImportException.NOT_FOUND_OR_AUTH_REQUIRED, 

363 HuggingFaceImportException.MODEL_NOT_FOUND, 

364 HuggingFaceImportException.FETCH_FAILED, 

365 } 

366 if isinstance(code, str) and code in known: 

367 raise HuggingFaceImportException(code, message)

369 @staticmethod 

370 def _build_hf_exception(error: str) -> HuggingFaceImportException: 

371 """Map a backend ``error`` field to a typed HuggingFaceImportException. 

372 

373 Some codes carry a payload separated by ``": "`` (``download_failed: foo.bin``, 

374 ``invalid_filename: ../etc``, ``registration_failed: ...``). Strip that for 

375 the ``error_code`` so callers can match on the prefix, but keep the full 

376 message human-readable. 

377 """ 

378 # download_failed: <file>, invalid_filename: <name>, registration_failed: <text> 

379 m = re.match(r"^([a-z_]+)(?:\s*:\s*(.*))?$", error or "") 

380 code = m.group(1) if m else (error or "unknown_error") 

381 return HuggingFaceImportException(code, error)

383 def _resolve_imported_model( 

384 self, hugging_face_model_id: str, status: dict 

385 ) -> model.Model: 

386 """Look up the registered Model after a successful import. 

387 

388 Mirrors the backend's name sanitiser (non-alphanumerics → ``_``) to 

389 find the right model entry, then returns the highest-version match. 

390 """ 

391 # Mirror the backend's parseModelId: drop scheme + host, drop query and 

392 # fragment, then keep only the first two non-empty path segments 

393 # ("owner/repo"). HF UI URLs like .../tree/main, .../blob/main/README.md, 

394 # .../resolve/main/... need their trailing segments stripped before we 

395 # take the repo name as the basis for the sanitised model name. 

396 repo = hugging_face_model_id 

397 for prefix in ("https://", "http://", "//"): 

398 if repo.startswith(prefix): 

399 repo = repo[len(prefix) :] 

400 break 

401 for host in ("huggingface.co/", "www.huggingface.co/", "hf.co/"): 

402 if repo.startswith(host): 

403 repo = repo[len(host) :] 

404 break 

405 repo = repo.split("?", 1)[0].split("#", 1)[0] 

406 kept_segments: list[str] = [] 

407 for seg in repo.split("/"): 

408 if not seg: 

409 continue 

410 kept_segments.append(seg) 

411 if len(kept_segments) == 2: 

412 break 

413 last = kept_segments[-1] if kept_segments else "" 

414 sanitised = re.sub(r"[^A-Za-z0-9]", "_", last) 

415 # The status payload doesn't include the registered version directly, 

416 # so look up the latest matching name. The HF importer always registers 

417 # at MAX(version)+1, so the most recent get_models() entry is ours. 

418 candidates = self._model_api.get_models( 

419 sanitised, 

420 self.model_registry_id, 

421 shared_registry_project_name=self.shared_registry_project_name, 

422 ) 

423 if not candidates: 

424 raise HuggingFaceImportException( 

425 "registration_failed", 

426 f"Import for {hugging_face_model_id} reported COMPLETED but the " 

427 f"model {sanitised!r} is not visible in the registry.", 

428 ) 

429 # Sorted ascending by version typically; pick the highest just in case. 

430 return max(candidates, key=lambda m: getattr(m, "version", 0))
