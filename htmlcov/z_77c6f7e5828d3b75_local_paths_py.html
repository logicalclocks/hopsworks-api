23def _normalize_hopsfs_mount_path(path: str) -> str | None: 

24 """Project-relative HopsFS path for a FUSE-mounted ``path``, or ``None``. 

25 

26 Recognizes ``/hopsfs/<rest>`` (per-project mount in Jobs/Jupyter pods) 

27 and ``/mnt/hopsfs/<project>/<rest>`` (cluster-wide mount). 

28 """ 

29 if path.startswith(MODEL_REGISTRY.HOPSFS_MOUNT_PREFIX): 

30 return path.replace(MODEL_REGISTRY.HOPSFS_MOUNT_PREFIX, "", 1) 

31 # /mnt/hopsfs/ is rooted at HopsFS root; strip the next path segment 

32 # (project name) too so the result is project-relative. 

33 base = MODEL_REGISTRY.HOPSFS_MOUNT_PREFIX_BASE 

34 if path.startswith(base + "/"): 

35 rest = path[len(base) + 1 :] 

36 first_slash = rest.find("/") 

37 if first_slash != -1 and first_slash + 1 < len(rest): 

38 return rest[first_slash + 1 :] 

39 return None

42def _resolve_serving_file( 

43 local_engine, 

44 deployment_name: str, 

45 path: str | None, 

46 field_name: str, 

47 subdir: str, 

48 overwrite: bool = True, 

49 is_update: bool = False, 

50) -> str | None: 

51 """Resolve a user-supplied serving file path to an absolute HopsFS path. 

52 

53 Mirrors the classification used by 

54 [`ModelEngine._save`][hsml.engine.model_engine.ModelEngine._save]: 

55 

56 1. ``None`` → return unchanged. 

57 2. Absolute path that exists on disk → upload (or, if the path lives 

58 under a HopsFS FUSE mount, rewrite to ``/Projects/<p>/...`` without 

59 uploading). 

60 3. Relative path that exists when joined with cwd → resolve to the 

61 absolute form and treat as step 2. 

62 4. ``DatasetApi.exists(path)`` is true → it's a HopsFS path (absolute, 

63 project-relative, or ``hdfs://``); rewrite to the absolute 

64 ``/Projects/<p>/...`` form and return. 

65 5. Unresolved. 

66 On update, a bare basename is a backend-managed reference (a 

67 deployment fetched from the backend reports ``script_file`` / 

68 ``config_file`` that way) and is returned unchanged. 

69 A create, or an update with a path-like value (one containing a 

70 separator) that resolved to nothing, is a genuine mistake and raises 

71 ``ValueError`` instead of deferring to a later backend failure. 

72 

73 ``subdir`` is the per-role subfolder (``"predictor"``, ``"transformer"``, 

74 ``"config"``) under ``Deployments/<name>/resources/``; it prevents 

75 basename collisions across roles. 

76 

77 Extension and content validation is the backend's responsibility. 

78 """ 

79 if path is None: 

80 return path 

81 

82 project_name = client._get_instance()._project_name 

83 

84 # 1. Absolute local — may be a FUSE mount. 

85 if os.path.isabs(path) and os.path.exists(path): 

86 return _resolve_local_or_mount( 

87 local_engine, project_name, deployment_name, path, subdir, overwrite 

88 ) 

89 

90 # 2. Relative local — resolve via cwd, then run the same 

91 # FUSE-mount-vs-upload decision on the absolute form. This catches 

92 # `./predictor.py` from a Jobs/Jupyter pod whose cwd lives under 

93 # /hopsfs/... or /mnt/hopsfs/<project>/... 

94 abs_path = os.path.join(os.getcwd(), path) 

95 if os.path.exists(abs_path): 

96 return _resolve_local_or_mount( 

97 local_engine, project_name, deployment_name, abs_path, subdir, overwrite 

98 ) 

99 

100 # 3. HopsFS — absolute /Projects/..., project-relative, or hdfs:// URI. 

101 if local_engine._dataset_api.exists(path): 

102 if path.startswith("hdfs:/"): 

103 idx = path.find("/Projects") 

104 return path[idx:] if idx != -1 else path 

105 if path.startswith("/Projects/"): 

106 return path 

107 return f"/Projects/{project_name}/{path.lstrip('/')}" 

108 

109 if is_update and os.path.basename(path) == path: 

110 # Persisted deployment: the backend reports script_file / config_file 

111 # as a bare basename (no separator), which is neither a local file nor 

112 # resolvable via DatasetApi.exists. Pass it through untouched. A 

113 # path-like value that resolved to nothing (e.g. "./missing.py" or a 

114 # typoed "/Projects/..." path) is a genuine mistake, so fall through 

115 # and raise rather than defer the error to a later backend failure. 

116 return path 

117 

118 raise ValueError( 

119 f"Could not find {field_name}: '{path}' in the local filesystem " 

120 f"or in Hopsworks File System." 

121 )

124def _ensure_dataset_dir(local_engine, remote_dir: str) -> None: 

125 """Recursively create every missing segment of ``remote_dir``. 

126 

127 ``DatasetApi.mkdir`` does not create parents. 

128 The top-level dataset (first segment) is provisioned per project by the 

129 platform; it is never created here. 

130 """ 

131 segments = [s for s in remote_dir.split("/") if s] 

132 if len(segments) < 2: 

133 return 

134 ds_api = local_engine._dataset_api 

135 for i in range(2, len(segments) + 1): 

136 sub_path = "/".join(segments[:i]) 

137 if not ds_api.exists(sub_path): 

138 local_engine._mkdir(sub_path)

141def _resolve_local_or_mount( 

142 local_engine, project_name, deployment_name, abs_path, subdir, overwrite 

143): 

144 """Rewrite or upload an absolute local path and return the HopsFS path. 

145 

146 If ``abs_path`` is a FUSE mount, rewrite it to ``/Projects/<p>/...``. 

147 Otherwise upload it under 

148 ``Deployments/<deployment_name>/resources/<subdir>/``. 

149 """ 

150 # Normalize away `./` and `../` segments that `os.path.join(cwd, path)` 

151 # leaves intact. The upload path stays in OS-native form (Windows uses 

152 # backslashes); the FUSE-mount check needs forward slashes, so use a 

153 # local variant for that comparison only. 

154 abs_path = os.path.normpath(abs_path) 

155 mount_relative = _normalize_hopsfs_mount_path(abs_path.replace(os.sep, "/")) 

156 if mount_relative is not None: 

157 return f"/Projects/{project_name}/{mount_relative}" 

158 

159 remote_dir = ( 

160 f"{MODEL_SERVING.DEPLOYMENTS_DATASET}/{deployment_name}/" 

161 f"{MODEL_SERVING.DEPLOYMENT_RESOURCES_DIR}/{subdir}" 

162 ) 

163 _ensure_dataset_dir(local_engine, remote_dir) 

164 local_engine._upload(abs_path, remote_dir, overwrite=overwrite) 

165 return f"/Projects/{project_name}/{remote_dir}/{os.path.basename(abs_path)}"
