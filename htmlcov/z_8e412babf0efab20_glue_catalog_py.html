16"""AWS Glue Data Catalog integration shared across table formats. 

17 

18The Glue connector is reused by the Iceberg, Delta and Hudi engines. 

19For Iceberg the catalog owns the table's current-metadata pointer, so writes 

20and reads are mediated by the catalog. 

21For Delta and Hudi the on-path log/timeline stays authoritative; the catalog is 

22a discoverability mirror that is registered on create and synced on write, so 

23external engines (Athena, EMR, ...) can find the table by name. 

24 

25[`GlueCatalog`][hsfs.core.glue_catalog.GlueCatalog] owns the connector- and 

26data-source-derived information common to all three formats plus the shared 

27Spark-session orchestration (credentials and catalog configuration). 

28The format-specific catalog mechanics (how reads, writes, merges and schema 

29evolution use the catalog) live in each engine. 

30""" 

31

46class GlueCatalog: 

47 """Glue Data Catalog binding for a single feature group. 

48 

49 Holds the feature group and its Glue connector and derives the catalog name, 

50 database, table and warehouse from them. 

51 Obtain one with 

52 [`GlueCatalog._for_feature_group`][hsfs.core.glue_catalog.GlueCatalog._for_feature_group], 

53 which returns `None` for feature groups that are not Glue-backed. 

54 """ 

55 

56 def __init__(self, feature_group: FeatureGroup, connector: sc.StorageConnector): 

57 self._feature_group = feature_group 

58 self._connector = connector 

59 

60 @classmethod 

61 def _for_feature_group(cls, feature_group: FeatureGroup) -> GlueCatalog | None: 

62 """Return a `GlueCatalog` for the feature group, or None if it is not Glue-backed. 

63 

64 Parameters: 

65 feature_group: The feature group whose storage connector to inspect. 

66 

67 Returns: 

68 A `GlueCatalog` bound to the feature group, or `None` when the feature group has a different connector or none. 

69 """ 

70 from hsfs import storage_connector as sc 

71 

72 connector = feature_group.storage_connector 

73 if connector is not None and connector.type == sc.StorageConnector.GLUE: 

74 return cls(feature_group, connector) 

75 return None 

76 

77 @property 

78 def connector(self) -> sc.StorageConnector: 

79 """The Glue storage connector backing the feature group.""" 

80 return self._connector 

81 

82 @property 

83 def catalog_name(self) -> str: 

84 """The Spark catalog name used for the Glue Data Catalog.""" 

85 return self._connector.DEFAULT_CATALOG_NAME 

86 

87 @property 

88 def database_and_table(self) -> tuple[str | None, str]: 

89 """The Glue `(database, table)` for the feature group. 

90 

91 The data source holds the actual Glue database and table, so it takes 

92 precedence over the connector's `database` (which is the Hopsworks 

93 default database, not the Glue one). 

94 The table falls back to `<name>_<version>` when the data source has none. 

95 """ 

96 data_source = self._feature_group.data_source 

97 database = ( 

98 data_source.database if data_source and data_source.database else None 

99 ) or self._connector.database 

100 table = data_source.table if data_source and data_source.table else None 

101 if not table: 

102 table = self._feature_group.get_fg_name() 

103 return database, table 

104 

105 @property 

106 def identifier(self) -> str: 

107 """The `<database>.<table>` Glue Data Catalog identifier. 

108 

109 Raises: 

110 hopsworks.client.exceptions.FeatureStoreException: If neither the data source nor the connector provides a database. 

111 """ 

112 from hopsworks_common.client.exceptions import FeatureStoreException 

113 

114 database, table = self.database_and_table 

115 if not database: 

116 raise FeatureStoreException( 

117 "Glue feature groups require a database on the data source or connector." 

118 ) 

119 return f"{database}.{table}" 

120 

121 @property 

122 def qualified_name(self) -> str: 

123 """The catalog-qualified table name, `<catalog>.<database>.<table>`.""" 

124 return f"{self.catalog_name}.{self.identifier}" 

125 

126 @property 

127 def warehouse(self) -> str | None: 

128 """The S3 warehouse root for the Glue catalog. 

129 

130 A Glue warehouse is the root under which tables are laid out as 

131 `<warehouse>/<database>.db/<table>`, so the feature group's own location 

132 (which already includes the `<database>.db/<table>` suffix) is trimmed 

133 back to that root. 

134 The warehouse is only used when the catalog creates a new table; for an 

135 existing table the catalog uses the location recorded in its metadata. 

136 """ 

137 location = self._feature_group.location 

138 if not location: 

139 return None 

140 # Use the same database resolution as the identifier (data source first, 

141 # then connector default), so the suffix to trim matches the one used to 

142 # lay the table out under the warehouse. 

143 database, _ = self.database_and_table 

144 suffix = f"/{database}.db/" if database else None 

145 if suffix and suffix in location: 

146 return location[: location.index(suffix)] 

147 return location

169 def _set_jvm_credentials(self, spark_context) -> None: 

170 """Push the connector's AWS credentials into the driver JVM's SDK system properties. 

171 

172 The Glue metadata client authenticates through the AWS SDK default 

173 credentials chain rather than format-specific credential options, and 

174 the metadata call runs in the driver JVM, so the credentials are set as 

175 system properties which the default chain's 

176 `SystemPropertyCredentialsProvider` resolves first. 

177 

178 Warning: Process-global 

179 JVM system properties are shared across the process, so when several 

180 Glue connectors are used in one session the last one set wins for the 

181 Glue metadata client. 

182 

183 Does nothing in Spark Connect mode (no JVM bridge) or when the connector 

184 has no static credentials (the default chain then falls back to the 

185 instance/role credentials). 

186 

187 Parameters: 

188 spark_context: The Spark context providing the JVM bridge, or `None` under Spark Connect. 

189 """ 

190 if spark_context is None: 

191 return 

192 connector = self._connector 

193 if not (connector.access_key and connector.secret_key): 

194 return 

195 jvm = spark_context._jvm 

196 jvm.java.lang.System.setProperty("aws.accessKeyId", connector.access_key) 

197 jvm.java.lang.System.setProperty("aws.secretAccessKey", connector.secret_key) 

198 if connector.session_token: 

199 jvm.java.lang.System.setProperty( 

200 "aws.sessionToken", connector.session_token 

201 ) 

202 if connector.region: 

203 jvm.java.lang.System.setProperty("aws.region", connector.region) 

204 _logger.debug("Set AWS JVM system properties for the Glue catalog client")

206 def _glue_client(self): 

207 """Build a boto3 Glue client authenticated with the connector's credentials. 

208 

209 The static access/secret keys are used when present, otherwise the 

210 client falls back to the AWS default credential chain (instance/role 

211 credentials), mirroring the JVM-side behaviour in 

212 [`_set_jvm_credentials`][hsfs.core.glue_catalog.GlueCatalog._set_jvm_credentials]. 

213 

214 Returns: 

215 A boto3 Glue client for the connector's region. 

216 """ 

217 import boto3 

218 

219 connector = self._connector 

220 kwargs: dict[str, Any] = {} 

221 if connector.region: 

222 kwargs["region_name"] = connector.region 

223 if connector.access_key and connector.secret_key: 

224 kwargs["aws_access_key_id"] = connector.access_key 

225 kwargs["aws_secret_access_key"] = connector.secret_key 

226 if connector.session_token: 

227 kwargs["aws_session_token"] = connector.session_token 

228 return boto3.client("glue", **kwargs)

230 def _register_delta_table(self, location: str) -> None: 

231 """Register the feature group's Delta table in the Glue Data Catalog. 

232 

233 Creates (or updates) a `name -> location` entry through the AWS Glue API 

234 so external engines can discover the table; the on-path Delta log stays 

235 authoritative. 

236 This deliberately avoids Spark SQL DDL through a named Delta catalog: 

237 Delta's `DeltaCatalog` is a `DelegatingCatalogExtension` that only works 

238 as the session catalog, so registering through a named catalog raises a 

239 null-delegate `NullPointerException` during analysis. 

240 

241 The table is marked as Delta (`table_type=DELTA`, 

242 `spark.sql.sources.provider=delta`) and its columns are taken from the 

243 feature group schema, whose types are already offline (Hive) SQL types 

244 that the Glue Data Catalog accepts verbatim. 

245 

246 Parameters: 

247 location: The absolute S3 location of the Delta table. 

248 """ 

249 from botocore.exceptions import ClientError 

250 

251 database, table = self.database_and_table 

252 columns = [ 

253 {"Name": feature.name, "Type": feature.type} 

254 for feature in self._feature_group.features 

255 ] 

256 table_input = { 

257 "Name": table, 

258 "TableType": "EXTERNAL_TABLE", 

259 "Parameters": { 

260 "table_type": "DELTA", 

261 "spark.sql.sources.provider": "delta", 

262 "EXTERNAL": "TRUE", 

263 }, 

264 "StorageDescriptor": { 

265 "Columns": columns, 

266 "Location": location, 

267 }, 

268 } 

269 

270 client = self._glue_client() 

271 _logger.debug(f"Registering Delta table in Glue as {database}.{table}") 

272 try: 

273 client.create_table(DatabaseName=database, TableInput=table_input) 

274 except ClientError as e: 

275 if e.response.get("Error", {}).get("Code") != "AlreadyExistsException": 

276 raise 

277 # Keep the existing entry current (e.g. schema changes on later writes). 

278 client.update_table(DatabaseName=database, TableInput=table_input)

280 def _configure_spark_session( 

281 self, 

282 spark_session, 

283 spark_context, 

284 default_catalog_impl: str, 

285 catalog_properties: dict[str, str] | None = None, 

286 ) -> None: 

287 """Prepare the Spark session to use the Glue catalog. 

288 

289 Sets the Glue metadata credentials on the driver JVM and configures 

290 `spark.sql.catalog.<name>.*` from the catalog properties. 

291 The catalog implementation comes from the properties' `impl` entry when 

292 present, otherwise from `default_catalog_impl` (Iceberg's `SparkCatalog` 

293 or Delta's `DeltaCatalog`), and only when the key is not already set. 

294 

295 Parameters: 

296 spark_session: The Spark session to configure. 

297 spark_context: The Spark context for setting JVM credentials, or `None` under Spark Connect. 

298 default_catalog_impl: The catalog implementation class to use when none is configured. 

299 catalog_properties: The catalog properties to apply; defaults to the connector's catalog options. 

300 """ 

301 self._set_jvm_credentials(spark_context) 

302 

303 properties = dict( 

304 self._catalog_properties() 

305 if catalog_properties is None 

306 else catalog_properties 

307 ) 

308 base_key = f"spark.sql.catalog.{self.catalog_name}" 

309 impl = properties.pop("impl", None) 

310 if impl is not None: 

311 spark_session.conf.set(base_key, impl) 

312 elif spark_session.conf.get(base_key, None) is None: 

313 spark_session.conf.set(base_key, default_catalog_impl) 

314 for prop, value in properties.items(): 

315 spark_session.conf.set(f"{base_key}.{prop}", value)

317 def _iceberg_write_options( 

318 self, 

319 write_options: dict[str, Any], 

320 catalog_option: str, 

321 identifier_option: str, 

322 property_prefix: str, 

323 ) -> dict[str, Any]: 

324 """Augment Iceberg write options to commit through the Glue Data Catalog. 

325 

326 Sets the catalog name, the `<database>.<table>` identifier and the 

327 catalog properties (each under `property_prefix`), so that a write 

328 registers the table in the Glue Data Catalog while the data stays on S3. 

329 

330 Parameters: 

331 write_options: The write options to augment (copied, not mutated). 

332 catalog_option: The write-option key naming the catalog (Iceberg's `iceberg.catalog`). 

333 identifier_option: The write-option key for the table identifier. 

334 property_prefix: The prefix under which catalog properties are passed. 

335 

336 Returns: 

337 A new options dict augmented with the Glue catalog configuration. 

338 """ 

339 write_options = dict(write_options) 

340 write_options[catalog_option] = self.catalog_name 

341 write_options[identifier_option] = self.identifier 

342 for key, value in self._catalog_properties().items(): 

343 write_options[f"{property_prefix}{key}"] = value 

344 _logger.debug( 

345 f"Routing Glue feature group {self._feature_group.name} " 

346 f"v{self._feature_group.version} through the Glue Data Catalog " 

347 f"as {write_options[identifier_option]}" 

348 ) 

349 return write_options
