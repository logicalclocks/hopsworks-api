55def _warmup_barrier_wait(barrier, timeout): 

56 """Worker-side warmup task: rendezvous on the barrier, then return. 

57 

58 Module-level so it pickles cleanly under spawn / forkserver start methods. 

59 Submitting `n_processes` instances of this task with a `Barrier(n_processes)` 

60 forces ProcessPoolExecutor to spawn every configured worker before any task 

61 can complete, turning a lazily-spawned pool into a fully-warm one. 

62 

63 The timeout bounds the rendezvous: if the pool cannot scale to the full 

64 worker count (resource limits, failed spawns) the barrier breaks, every 

65 parked worker unblocks with `BrokenBarrierError`, and the parent surfaces 

66 the failure instead of hanging forever. 

67 """ 

68 barrier.wait(timeout)

258 @staticmethod 

259 def _force_full_spawn(n_processes: int, timeout: float) -> None: 

260 """Force the pool to spawn all `n_processes` workers before returning. 

261 

262 ProcessPoolExecutor spawns workers lazily on first submit, so submitting 

263 `n_processes` tasks that all rendezvous on a shared Barrier forces every 

264 worker to start: no task can return until all of them reach the barrier. 

265 The barrier and the result collection are bounded by `timeout`, so a pool 

266 that cannot reach `n_processes` workers breaks the barrier in every parked 

267 worker instead of deadlocking the caller. 

268 

269 Raises: 

270 FeatureStoreException: If the pool fails to warm `n_processes` workers within `timeout` seconds. 

271 """ 

272 pool = TransformationFunctionEngine.__process_pool 

273 manager = multiprocessing.Manager() 

274 try: 

275 barrier = manager.Barrier(n_processes) 

276 futures = [ 

277 pool.submit(_warmup_barrier_wait, barrier, timeout) 

278 for _ in range(n_processes) 

279 ] 

280 try: 

281 for future in futures: 

282 future.result(timeout=timeout) 

283 except Exception as e: 

284 TransformationFunctionEngine._shutdown_process_pool( 

285 wait=False, cancel_futures=True 

286 ) 

287 raise exceptions.FeatureStoreException( 

288 f"Worker pool failed to warm up {n_processes} workers within " 

289 f"{timeout}s. The pool has been discarded; check process and " 

290 "memory limits, or lower n_processes." 

291 ) from e 

292 finally: 

293 manager.shutdown()

297 """Resolve the multiprocessing start method for the TF worker pool. 

298

302 to on Linux; macOS and Windows default to `spawn`. 

303

305 default with `fork`, `forkserver`, or `spawn` (except on Windows, where 

306 `fork` is unavailable).

309 return "spawn" 

310 default = "fork" if sys.platform.startswith("linux") else "spawn"

313 return default

315 return override

317 # pool create / swap, so without the latch a persistent bad env value 

318 # would flood the log of a long-running process.

320 _logger.warning(

324 default,

335 """Validate that all arguments required to execute the transformation functions are present in the passed data or request parameters. 

336 

337 Parameters: 

338 execution_graph: The transformation DAG containing the transformation functions to validate. 

339 data: The dataframe or dictionary to validate the transformation functions against. 

340 request_parameters: Request parameters to validate the transformation functions against. 

341 

342 Raises: 

343 exceptions.TransformationFunctionException: If the arguments required to execute the transformation functions are not present in the passed data or request parameters. 

344 """ 

345 transformation_function_output_feature = set() 

346 # Fast-path: dict/list skip the engine check entirely (avoids engine._get_instance() overhead) 

347 if isinstance(data, (dict, list)): 

348 is_dataframe = False 

349 else: 

350 is_dataframe = engine._get_instance()._check_supported_dataframe(data) 

351 if not is_dataframe: 

352 raise exceptions.FeatureStoreException( 

353 f"Dataframe type {type(data)} not supported in the engine." 

354 ) 

355 

356 # A list of per-row request parameters is only valid against row-shaped 

357 # data (a DataFrame or list of dicts), and must match its row count. A 

358 # single dict is one row, so `len(data)` (its key count) is not the row 

359 # count and must not be used here. 

360 if isinstance(request_parameters, list): 

361 n_rows = 1 if isinstance(data, dict) else len(data) 

362 if len(request_parameters) != n_rows: 

363 raise exceptions.TransformationFunctionException( 

364 "Request Parameters should be a Dictionary, None, empty or be a list having the same length as the number of rows in the data provided." 

365 ) 

366 

367 # Compute available keys per "row" so a feature counts as available 

368 # for validation only when it is reachable on every row at execution. 

369 # Per-row execution merges request_parameters into the data row 

370 # before invoking the UDF, so the effective per-row available keys 

371 # are (row data keys) UNION (row request_parameter keys). A feature 

372 # missing on some rows but present via request_parameters on others 

373 # must still pass validation as long as every row has at least one 

374 # source for it. 

375 if is_dataframe: 

376 data_keys_per_row: list[set[str]] = [set(data.columns)] 

377 elif isinstance(data, list): 

378 data_keys_per_row = [set(row.keys()) for row in data] or [set()] 

379 else: 

380 data_keys_per_row = [set(data.keys())] 

381 

382 if request_parameters: 

383 if isinstance(request_parameters, list): 

384 rp_keys_per_row: list[set[str]] = [ 

385 set(rp.keys()) for rp in request_parameters 

386 ] or [set()] 

387 else: 

388 rp_keys_per_row = [set(request_parameters.keys())] 

389 else: 

390 rp_keys_per_row = [set()] 

391 

392 # For DataFrame input, _update_request_parameter_data concatenates 

393 # list-shaped request_parameters into a DataFrame with the union of 

394 # keys (missing values fill as NaN), so on the DataFrame path the 

395 # "available" set is the union of every row's request_parameter 

396 # keys, not their intersection. 

397 if is_dataframe and len(rp_keys_per_row) > 1: 

398 rp_keys_per_row = [set().union(*rp_keys_per_row)] 

399 

400 # Broadcast singletons to match list lengths, then per-row union and 

401 # cross-row intersection. 

402 n_rows = max(len(data_keys_per_row), len(rp_keys_per_row)) 

403 if len(data_keys_per_row) == 1: 

404 data_keys_per_row = data_keys_per_row * n_rows 

405 if len(rp_keys_per_row) == 1: 

406 rp_keys_per_row = rp_keys_per_row * n_rows 

407 per_row_available = [ 

408 d | r for d, r in zip(data_keys_per_row, rp_keys_per_row, strict=False) 

409 ] 

410 available_features: set[str] = ( 

411 per_row_available[0].copy() if per_row_available else set() 

412 ) 

413 for row_keys in per_row_available[1:]: 

414 available_features.intersection_update(row_keys) 

415 

416 # Also gather the cross-row available request-parameter keys for the 

417 # prefix-fallback step below. Same intersection rule: a prefixed 

418 # lookup only counts as resolvable when every row supplies the key. 

419 rp_available: set[str] = rp_keys_per_row[0].copy() if rp_keys_per_row else set() 

420 for rp in rp_keys_per_row[1:]: 

421 rp_available.intersection_update(rp) 

422 

423 for tf in execution_graph.nodes: 

424 missing_features = ( 

425 set(tf.hopsworks_udf.transformation_features) - available_features 

426 ) 

427 if tf.hopsworks_udf.feature_name_prefix and rp_available: 

428 missing_features = missing_features - { 

429 tf.hopsworks_udf.feature_name_prefix + feature 

430 for feature in rp_available 

431 } 

432 missing_features = missing_features - transformation_function_output_feature 

433 

434 if missing_features: 

435 raise exceptions.TransformationFunctionException( 

436 message=f"The following feature(s): `{', '.join(missing_features)}`, required for the transformation function '{tf.hopsworks_udf.function_name}' are not available.", 

437 missing_features=missing_features, 

438 transformation_function_name=tf.hopsworks_udf.function_name, 

439 transformation_type=tf.transformation_type.value, 

440 ) 

441 transformation_function_output_feature.update( 

442 tf.hopsworks_udf.output_column_names 

443 )

559 @staticmethod 

560 def _transformed_column_layout( 

561 execution_graph: TransformationExecutionDAG, 

562 data_columns: list[str], 

563 expected_features: set[str] | None, 

564 ) -> tuple[set[str], list[str]]: 

565 """Compute the dropped features and the final column order for a transform. 

566 

567 Drops are applied as a single projection after every transformation has 

568 run, not interleaved, so this layout describes the result: the original 

569 columns that survive, followed by the transformation outputs in 

570 topological order. 

571 

572 Parameters: 

573 execution_graph: The transformation DAG. 

574 data_columns: Columns present in the input (empty when only the dropped set is needed). 

575 expected_features: Features kept even when a transformation drops them, so a feature shared across feature groups is not removed. 

576 

577 Returns: 

578 The set of dropped feature names and the ordered list of resulting columns. 

579 """ 

580 dropped_features: set[str] = set() 

581 tf_output_cols: list[str] = [] 

582 tf_output_set: set[str] = set() 

583 for tf in execution_graph.nodes: 

584 udf = tf.hopsworks_udf 

585 if udf.dropped_features: 

586 dropped_features.update( 

587 {f for f in udf.dropped_features if f not in expected_features} 

588 if expected_features is not None 

589 else udf.dropped_features 

590 ) 

591 for col in udf.output_column_names: 

592 tf_output_set.add(col) 

593 tf_output_cols.append(col) 

594 column_order = [ 

595 c for c in data_columns if c not in tf_output_set 

596 ] + tf_output_cols 

597 return dropped_features, column_order

864 @staticmethod 

865 def _schedule_dag( 

866 execution_graph: TransformationExecutionDAG, 

867 submit: Callable[[transformation_function.TransformationFunction], Any], 

868 collect: Callable[[transformation_function.TransformationFunction, Any], None], 

869 ) -> None: 

870 """Drive a streaming DAG schedule over the process pool. 

871 

872 ``submit(tf)`` submits one transformation's work and returns its future; 

873 ``collect(tf, result)`` consumes that future's result. A transformation is 

874 submitted as soon as its predecessors complete (``wait(FIRST_COMPLETED)``), 

875 so independent branches run concurrently and a dependent starts the moment 

876 it unblocks. On any worker failure the pending futures are cancelled, 

877 in-flight ones drained (so no worker is mid-read of state the caller is 

878 about to release), and the singleton pool discarded before re-raising. 

879 """ 

880 id_to_tf = {id(tf): tf for tf in execution_graph.nodes} 

881 sorter = execution_graph._new_topological_sorter() 

882 future_to_tf: dict[Any, int] = {} 

883 try: 

884 while sorter.is_active(): 

885 for tf_id in sorter.get_ready(): 

886 future_to_tf[submit(id_to_tf[tf_id])] = tf_id 

887 done, _ = wait(future_to_tf.keys(), return_when=FIRST_COMPLETED) 

888 for future in done: 

889 tf_id = future_to_tf.pop(future) 

890 collect(id_to_tf[tf_id], future.result()) 

891 sorter.done(tf_id) 

892 except (Exception, BrokenProcessPool): 

893 for f in future_to_tf: 

894 f.cancel() 

895 if future_to_tf: 

896 wait(future_to_tf.keys(), return_when=ALL_COMPLETED) 

897 TransformationFunctionEngine._shutdown_process_pool( 

898 wait=False, cancel_futures=True 

899 ) 

900 raise

928 def _write_to_shared_memory( 

929 dataframe: pd.DataFrame, 

930 ) -> tuple[shared_memory.SharedMemory, int, bool]: 

931 """Serialize a DataFrame to Arrow IPC in shared memory. 

932 

933 Returns (shm, buffer_size, is_polars). Caller must call 

934 shm.close() and shm.unlink() after all workers are done. 

935 """ 

936 is_polars = HAS_POLARS and isinstance(dataframe, pl.DataFrame) 

937 if is_polars: 

938 table = dataframe.to_arrow() 

939 else: 

940 table = pa.Table.from_pandas(dataframe, preserve_index=False) 

941 

942 sink = pa.BufferOutputStream() 

943 writer = ipc.new_stream(sink, table.schema) 

944 writer.write_table(table) 

945 writer.close() 

946 buf = sink.getvalue() # pa.Buffer, no Python bytes copy 

947

953 # segment exists but the caller will never see it. Release it here 

954 # so we don't leak /dev/shm/psm_* segments on partial failure. 

955 # Best-effort cleanup that must NOT mask the original failure, so each 

956 # step is wrapped independently and any cleanup error is logged. 

957 try: 

958 shm.close() 

959 except Exception as cleanup_exc: 

960 _logger.warning( 

961 "shm.close() failed during shared-memory allocation rollback: %s", 

962 cleanup_exc, 

963 ) 

964 try: 

965 shm.unlink() 

966 except Exception as cleanup_exc: 

967 _logger.warning( 

968 "shm.unlink() failed during shared-memory allocation rollback: %s", 

969 cleanup_exc, 

970 ) 

971 raise 

972 # Do NOT close here: on Windows the mapping is destroyed when the 

973 # last handle is closed. The caller must keep this object alive and 

974 # call close()/unlink() after all workers are done. 

975 return shm, buf.size, is_polars

1377 interleave in one fused pass over the DAG.

1380 see the test rows and there is no leakage.

1384 feature_view_obj: The feature view whose transformation functions are fit and applied.

1403 # the execution DAG's resolution rule: a transformation that overwrites
