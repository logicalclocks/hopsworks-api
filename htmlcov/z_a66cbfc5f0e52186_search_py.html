1"""``hops search`` — feature-store search, project-scoped or global. 

2 

3Wraps ``project.get_search_api().feature_store(...)``: a single REST call 

4against the OpenSearch ``featurestore`` index that returns matching 

5feature groups, feature views, training datasets, and features. Project 

6scope is the default; ``--global`` searches every project the caller has 

7access to. 

8"""

21@click.group("search") 

22def search_group() -> None: 

23 """Search the Hopsworks feature store (project-scoped by default).""" 

24 

25 

26@search_group.command("ls") 

27@click.argument("term", required=False) 

28@click.option( 

29 "--global", 

30 "global_search", 

31 is_flag=True, 

32 help="Search across all projects the caller can see (default: current project only).", 

33) 

34@click.option( 

35 "--type", 

36 "doc_type", 

37 type=click.Choice(_DOC_TYPES, case_sensitive=False), 

38 default="all", 

39 show_default=True, 

40 help="Restrict results to one entity type.", 

41) 

42@click.option( 

43 "--keyword", 

44 "keywords", 

45 multiple=True, 

46 help="Filter by keyword. Repeat for multiple keywords.", 

47) 

48@click.option( 

49 "--tag", 

50 "tags", 

51 multiple=True, 

52 help='Filter by tag, format "name:key=value". Repeat for multiple tags.', 

53) 

54@click.option("--limit", type=int, default=20, show_default=True, help="Max results.") 

55@click.pass_context 

56def search_ls( 

57 ctx: click.Context, 

58 term: str | None, 

59 global_search: bool, 

60 doc_type: str, 

61 keywords: tuple[str, ...], 

62 tags: tuple[str, ...], 

63 limit: int, 

64) -> None: 

65 """Search for entities matching TERM. 

66 

67 Args: 

68 ctx: Click context. 

69 term: Search string. May be omitted if ``--keyword`` or ``--tag`` 

70 is given (the SDK requires at least one of the three). 

71 global_search: When True, search across all projects. 

72 doc_type: One of ``all``, ``feature_group``, ``feature_view``, 

73 ``training_dataset``, ``feature``. 

74 keywords: Repeatable ``--keyword`` filter. 

75 tags: Repeatable ``--tag name:key=value`` filter. 

76 limit: Page size cap. 

77 """ 

78 if not term and not keywords and not tags: 

79 raise click.UsageError( 

80 "Provide a search TERM, or at least one --keyword / --tag filter." 

81 ) 

82 

83 parsed_tags = [_parse_tag(t) for t in tags] 

84 

85 project = session.get_project(ctx) 

86 api = project.get_search_api() 

87 

88 common_kwargs: dict[str, Any] = { 

89 "search_term": term, 

90 "keyword_filter": list(keywords) or None, 

91 "tag_filter": parsed_tags or None, 

92 "limit": limit, 

93 "global_search": global_search, 

94 } 

95 

96 try: 

97 if doc_type == "all": 

98 result = api.feature_store(**common_kwargs) 

99 rows = ( 

100 [_row("feature_group", r) for r in result.feature_groups] 

101 + [_row("feature_view", r) for r in result.feature_views] 

102 + [_row("training_dataset", r) for r in result.training_datasets] 

103 + [_row("feature", r) for r in result.features] 

104 ) 

105 elif doc_type == "feature_group": 

106 rows = [ 

107 _row("feature_group", r) for r in api.feature_groups(**common_kwargs) 

108 ] 

109 elif doc_type == "feature_view": 

110 rows = [_row("feature_view", r) for r in api.feature_views(**common_kwargs)] 

111 elif doc_type == "training_dataset": 

112 rows = [ 

113 _row("training_dataset", r) 

114 for r in api.training_datasets(**common_kwargs) 

115 ] 

116 else: # feature 

117 rows = [_row("feature", r) for r in api.features(**common_kwargs)] 

118 except Exception as exc: # noqa: BLE001 

119 raise click.ClickException(f"Search failed: {exc}") from exc 

120 

121 if not rows: 

122 scope = "all projects" if global_search else "current project" 

123 output.info("No results in %s.", scope) 

124 return 

125 

126 output.print_table(["KIND", "NAME", "VERSION", "PROJECT", "DESCRIPTION"], rows)

129def _row(kind: str, item: Any) -> list[Any]: 

130 """Flatten a SearchResultItem into a fixed-width table row. 

131 

132 Features don't have a ``version`` attribute; fall back to ``-``. 

133 Description is truncated to keep the table readable on narrow shells. 

134 """ 

135 name = getattr(item, "name", "?") 

136 version = getattr(item, "version", None) 

137 project_obj = getattr(item, "project", None) 

138 project_name = getattr(project_obj, "name", "-") if project_obj else "-" 

139 desc = (getattr(item, "description", None) or "").splitlines()[0:1] 

140 desc_text = (desc[0] if desc else "")[:60] 

141 return [kind, name, str(version) if version else "-", project_name, desc_text]

144def _parse_tag(spec: str) -> dict[str, str]: 

145 """Parse ``name:key=value`` into the {name, key, value} dict the SDK expects. 

146 

147 Raise ``click.BadParameter`` (rather than ``ValueError``) so the CLI 

148 user sees a clean message instead of a Python traceback. 

149 """ 

150 if ":" not in spec or "=" not in spec: 

151 raise click.BadParameter( 

152 f"--tag must be 'name:key=value', got {spec!r}", param_hint="--tag" 

153 ) 

154 name, rest = spec.split(":", 1) 

155 key, value = rest.split("=", 1) 

156 if not (name.strip() and key.strip() and value.strip()): 

157 raise click.BadParameter( 

158 f"--tag fields must be non-empty, got {spec!r}", param_hint="--tag" 

159 ) 

160 return {"name": name.strip(), "key": key.strip(), "value": value.strip()}
